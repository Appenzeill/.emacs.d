*** Eshell
#+BEGIN_SRC emacs-lisp
(defun eshell-here ()
  (interactive)
  (if (get-buffer "*eshell*")
  (kill-buffer-and-window)
   (let* ((height (/ (window-total-height) 3)))
    (split-window-vertically (- height))
    (other-window 1)
    (eshell "new")
    (insert (concat "ls"))
    (eshell-send-input))))
  (bind-key "C-`" 'eshell-here)
